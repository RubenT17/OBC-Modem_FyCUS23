364 float DS18B20_Decode_Temperature (GPIO_TypeDef* GPIOx, uint16_t GPIO_Pin, uint8_t *data)
365 {
366  float temperature = 0.0;
367  uint8_t temp_LSB = data[0];
368  uint8_t temp_MSB = data[1];
369  uint16_t u16_temp = ((temp_MSB << 8) | temp_LSB);
